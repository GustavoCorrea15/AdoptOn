# Sistema de Adoção Responsável - Dependências Globais

# Docker (obrigatório)
# Instalar Docker Desktop: https://www.docker.com/products/docker-desktop/

# Node.js (para desenvolvimento local)
# Instalar Node.js 18+: https://nodejs.org/

# Git (para versionamento)
# Instalar Git: https://git-scm.com/

# Expo CLI (para app mobile)
npm install -g @expo/cli

# Dependências opcionais para desenvolvimento
npm install -g nodemon
npm install -g concurrently

# Para Windows (opcional)
# Instalar Windows Subsystem for Linux (WSL2)
# wsl --install

# Para macOS (opcional)
# Instalar Homebrew: /bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
# brew install docker
# brew install node
# brew install git
